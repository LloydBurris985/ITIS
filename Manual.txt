Quick usage examples
Encode a file
Pythoneye = OdinsEye(start_mask=50000)
with open("photo.jpg", "rb") as f:
    coord = eye.encode(f.read())
print(coord)
# Save coord to JSON for later recovery
import json
with open("photo_coord.json", "w") as f:
    json.dump(coord, f)
Recover known file from coord
Pythonwith open("photo_coord.json") as f:
    coord = json.load(f)
recovered = eye.decode(coord)
with open("recovered_photo.jpg", "wb") as f:
    f.write(recovered)
